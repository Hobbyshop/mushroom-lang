### Example code
```
func main() {
    var name: string = "Bob"
    say_something(name, "Hello World")

    if (name == "Bob") {
        println("Hello Bob")
    }
}

func say_something(name: string, text: string) {
    println("${name} said, ${text}")
}

func do_maths(x: int, y: int) int {
    return x + y
}
```
